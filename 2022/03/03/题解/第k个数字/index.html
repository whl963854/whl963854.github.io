        int kthGrammar(int n, int k){
    int sign=1;
    while(1){
        if(k<=(1<<(n-1))){
            n--;
        }
        if(n&&k>(1<<(n-1))){
            k=k-(1<<(n-1));
            sign*=-1;
        }
        if(k==1){
            return sign>0?0:1;
        }
    }
}

        int kthGrammar(int N, unsigned int K) {
        K -= 1;
        K ^= K >> 1;
        K ^= K >> 2;
        K = (K & 0x11111111U) * 0x11111111U;
        return (K >> 28) & 1;
    }
