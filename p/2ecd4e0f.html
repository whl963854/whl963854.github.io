    基于xfx torm的学习开发和容器化打包部署

    stage 生成代码

    stage 可以帮助开发人员定义并产出规范的工程项目。利用 stage 我们可以用极快的速度生产：

    mkdir -p ./src/hello
cd ./src/hello
stage init
echo "get /hello 200 {greeting: string};" >> spec/hello.xidl
stage gen
go mod tidy
make build
./run serve

    编译后的 hello 是一个可以直接运行的服务器程序。可以通过 curl 命令验证服务器是否运行正常：

    curl http://localhost:8080/hello

    stage init 命令类似于 git init 命令，将一个目录初始化为规范工程，它产生了一个 spec/hello.xidl 文件，这个文件用来定义程序接口规格，默认情况下 stage 定义好了最基本的命令行接口。

    stage init

    然后向 hello.xidl 追加了一条 /hello 的定义，这是一种紧凑型的接口定义语言。

    echo "get /hello 200 {greeting: string};" >> spec/hello.xidl

    stage gen 命令会根据接口规格描述产生工程代码。

    stage gen

    注：
    需要配置完成go环境以后使用stage
    需要将stage.exe 加入到环境变量方便我们在其他文件夹内使用该命令
    make命令需要下载MinGW 并且配置make环境变量

    产生DAO代码

    go install ksogit.kingsoft.net/o/stage/v2@latest
# v2
go get ksogit.kingsoft.net/o/xfx@dev2

    新建一个用户相关 API 的工程：

    mkdir ./src/user
cd ./src/user
stage init

    然后往 spec/user.xidl 文件中追加创建用户的接口定义：

    schema user {
    id: string
    name: string
}

schema user_create_request {
    name: string
}

@name "create_user"
post /users
    . user_create_request
    200 user
;

    通过运行 stage gen 后，stage 产生了 NewCreateUserService 的代码，我们只要关注这个 service 进行编写相关的业务代码即可。 但是，这个例子中核心模型 user 是需要入库的，因此，我们还需要编写枯燥的 DAO 来辅助 service 对 user 进行相关的持久化操作。我们可以通过stage解决这个问题，stage 产生 DAO 代码是靠 @table 和 @column 两个注解来进行定义的：

    @table "tb_user;ORM(torm)"
schema user {
    @column "pk;ai"
    id: string

    name: string
}

schema user_create_request {
    name: string
}

@name "create_user"
post /users
    . user_create_request
    200 user
;

    @table 指示 user 采用 “tb_user” 作为表名。@column 指示了 id 是主键(pk)和自增长(ai)字段。 ORM(torm) 指定了要用的 ORM 库

    stage gen 针对这种定义主要做了三件事：

    一、将 DAO 相关的代码产生到工程目录下的 dao 子目录中。

├── user_delete.go
├── user_insert.go
├── user_model.go
├── user_order.go
├── user_select.go
├── user_set.go
├── user_update.go
├── user_where.go
└── util.go

    二、产生后端服务对接代码。

    // Code generated by xfx; DO NOT EDIT.

package ctx

import (
    "user/dao"
)

type backingServices struct {
    MysqlDB  *dao.MysqlDB
}

func newBackingServices(c *ModuleContext) *backingServices {
    var mysqlDB *dao.MysqlDB
    {
        writeWithoutPrepare, err := opts.GetBool("mysql-db-write-without-prepare", false)
        if err != nil {
            panic(err)
        }
        mysqlDB = dao.NewMysqlDB(
            c.AppContext.TORM("MYSQL"),
            dao.WithUniversalSQL(writeWithoutPrepare),
        )
    }
    return &backingServices{
        MysqlDB: mysqlDB,
    }
}

    利用生成的代码进行数据持久化

    Endpoint（端点）是一个网络服务的访问地址，客户端可以通过该地址与服务进行交互。在Web服务中，Endpoint通常指一个URL（Uniform Resource Locator，统一资源定位符），用于描述服务所在的网络位置以及可用的操作。Endpoint可以被认为是一个API的入口点，客户端可以使用它来访问服务并执行各种操作，例如获取数据、执行业务逻辑或进行状态更改。在SOAP协议中，Endpoint通常由一个地址（URL）和一个绑定（Binding）组成，它定义了如何对服务进行调用以及如何进行消息传输。在RESTful服务中，Endpoint通常由HTTP方法和URL组成，用于定义可以对资源执行的操作。

    我们在创建User对象的时候，把name初始化

    func NewCreateUserService(moduleCtx *ctx.ModuleContext) CreateUserFunc {
    return func(ctx context.Context, request *CreateUserRequest) (*CreateUserResponse, error) {
        //验证标签合法
        err := validator.Validate(request)
        if err != nil {
            return nil, err
        }
        out := NewCreateUserResponse()
        out.Name = request.Name
        return &out, nil
    }
}

    我们在Endpoint中调用数据库代码，将id和user持久化到数据库。

    func (h *CreateUser) Endpoint(app xfx.App) endpoint.Endpoint {
    callService := srv.NewCreateUserService(ctx.Get(app))
    return func(w http.ResponseWriter, r *http.Request) error {
        w.Header().Set("Access-Control-Allow-Origin", "*")
        in, err := parseCreateUserRequest(w, r)
        if err != nil {
            return rw.WriteError(w, r, el.Adapt(err))
        }
        in.EscapeStrings()
        out, err := callService(r.Context(), in)
        if err != nil {
            return rw.WriteError(w, r, err)
        }

        moduleCtx := ctx.Get(app)
        db:=moduleCtx.MysqlDB
        db.User().Insert(&dao.UserSet{
            Name:dao.String(out.Name),
        }).Exec(context.Background())
        userBuilder :=db.User().Select().Where(func(b *dao.UserWhereBuilder) {
            b.IdGT(0)
        })
        userid,_ := userBuilder.Count(context.Background())
        out.Id =fmt.Sprintf("%d", userid)
        return rw.WriteJson(w, 200, out)
    }
}

    将该服务打包成镜像

    配置Dockerfile文件

    ARG GIT_BRANCH
ARG BASE_IMAGE=hub-mirror.wps.cn/priopen/golang-nodejs-builder:1.18.7-alpine-v3
ARG USER
ARG PASSWORD

FROM ${BASE_IMAGE} as golang-build

# 根据CI配置克隆目录修改PROJECT_NAME参数
ARG PROJECT_NAME=user
ARG APP_NAME=user
ENV PROJECT_NAME=${PROJECT_NAME}
ENV APP_NAME=${APP_NAME}
ENV GIT_BRANCH=${GIT_BRANCH}
ENV TERM=xterm

RUN export GONOSUMDB=*.kingsoft.net
RUN export GOPRIVATE=*.kingsoft.net

RUN git config --global --add url."git@ksogit.kingsoft.net:".insteadof "https://ksogit.kingsoft.net/"
RUN git config --global credential.helper cache
RUN git config --global user.name "xxxxxxxxxx"
RUN git config --global user.password "xxxxxxxxxxxxxx"
RUN git config --global user.email "xxxxxx@kingsoft.com"

# 指定当前工作目录为 /opt/${PROJECT_NAME}/${APP_NAME}
WORKDIR /opt/${PROJECT_NAME}
# ADD ./docker /opt
COPY . /opt/${PROJECT_NAME}

RUN echo http://mirrors.aliyun.com/alpine/v3.15/main/ > /etc/apk/repositories && \
    echo http://mirrors.aliyun.com/alpine/v3.15/community/ >> /etc/apk/repositories


RUN source /opt/$PROJECT_NAME/env
# 执行 make 命令生成代码，并进行构建
RUN /usr/local/go/bin/go build -o user -v -ldflags " \
    -X user/cmds.GitHash=$(shell git rev-parse HEAD 2> /dev/null) \
    -X user/cmds.BuildTime=$(shell date "+%Y-%m-%dT%H:%M:%SZ")"


FROM hub-mirror.wps.cn/priopen/golang-runner:1.17.13-alpine

# 根据CI配置克隆目录修改PROJECT_NAME参数
ENV PROJECT_NAME=user
ENV APP_NAME=user

WORKDIR /opt/${PROJECT_NAME}

COPY --from=golang-build  /opt/${PROJECT_NAME}/docker ./docker
COPY --from=golang-build  /opt/${PROJECT_NAME}/${APP_NAME} ./
COPY --from=golang-build  /opt/${PROJECT_NAME}/env ./
RUN chmod +x ./docker/start.sh
RUN chmod +x ./docker/livenessProbe.sh
RUN chmod +x ./docker/readinessProbe.sh

RUN cp /opt/${PROJECT_NAME}/docker/livenessProbe.sh /usr/local/bin/livenessProbe.sh
RUN cp /opt/${PROJECT_NAME}/docker/readinessProbe.sh /usr/local/bin/readinessProbe.sh

RUN /bin/bash -c "source /opt/${PROJECT_NAME}/env"
EXPOSE 80 8081
ENTRYPOINT ["/bin/bash", "-c","/opt/${PROJECT_NAME}/docker/start.sh"]

    配置完Dockerfile以后，使用

    docker build -t user .

    构建一个名字为user的镜像,构建完成后，由该镜像构建一个容器

    docker run user -p 8080:8080

    注：
    构建过程中需要export GONOSUMDB=.kingsoft.net export GOPRIVATE=.kingsoft.net 否则会因为权限问题无法拉取xfx/o/wps 的仓库。
    需要更新一下env配置，RUN /bin/bash -c “source /opt/${PROJECT_NAME}/env” 否则运行会产生错误
